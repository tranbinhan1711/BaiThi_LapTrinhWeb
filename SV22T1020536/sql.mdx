PRINT N'--- TQ1: Vé chuyến SE1 từ Hà Nội (Node Bắc) ---';
SELECT t.TicketID, c.Name AS KhachHang, t.SeatNo, t.SeatClass, t.Price
FROM Tickets_North t
JOIN Customers c  ON t.CustomerID = c.CustomerID
JOIN Schedules s  ON t.ScheduleID = s.ScheduleID
JOIN Trains    tr ON s.TrainID    = tr.TrainID
JOIN Stations  og ON s.OriginID   = og.StationID
WHERE tr.TrainCode = 'SE1' AND og.StationName = N'Ga Hà Nội';

PRINT N'--- TQ2: Lịch trình tất cả chuyến tàu ---';
SELECT tr.TrainCode, og.StationName AS DiemDi, ds.StationName AS DiemDen,
       s.DepartureTime, s.ArrivalTime
FROM Schedules s
JOIN Trains   tr ON s.TrainID       = tr.TrainID
JOIN Stations og ON s.OriginID      = og.StationID
JOIN Stations ds ON s.DestinationID = ds.StationID
ORDER BY s.DepartureTime;

PRINT N'--- TQ3: Tổng số vé và doanh thu ---';
SELECT COUNT(*) AS TongVe, SUM(Price) AS TongDoanhThu FROM Tickets;

SELECT s.StudentID, s.Name, s.DOB, s.Gender, c.CampusName
FROM Students_North s
JOIN Campuses c ON s.CampusID = c.CampusID;

PRINT N'--- TQ2: Tất cả môn học toàn hệ thống ---';
SELECT co.CourseID, co.CourseName, co.Credits, ca.CampusName
FROM Courses co
JOIN Faculties f  ON co.FacultyID = f.FacultyID
JOIN Campuses  ca ON f.CampusID   = ca.CampusID
ORDER BY co.CourseID;

PRINT N'--- TQ3: Điểm trung bình toàn hệ thống ---';
SELECT
    AVG(CASE WHEN RegionID = 1 THEN Grade END) AS DiemTB_Bac,
    AVG(CASE WHEN RegionID = 2 THEN Grade END) AS DiemTB_Trung,
    AVG(CASE WHEN RegionID = 3 THEN Grade END) AS DiemTB_Nam,
    AVG(Grade) AS DiemTB_ToanHeThong
FROM Enrollments;

PRINT N'--- TQ4: Bảng điểm sinh viên miền Nam ---';
SELECT s.Name AS SinhVien, co.CourseName AS MonHoc, e.Grade AS Diem
FROM Enrollments_South e
JOIN Students s  ON e.StudentID = s.StudentID
JOIN Classes  cl ON e.ClassID   = cl.ClassID
JOIN Courses  co ON cl.CourseID = co.CourseID
ORDER BY s.Name, co.CourseName;
